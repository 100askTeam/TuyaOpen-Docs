人脸跟随

本文档详细介绍如何使用 Hopenet 模型实现人脸姿态估计和机械臂跟随功能。通过检测人脸的三维角度（yaw、pitch、roll），计算偏差并控制机械臂实时跟踪人脸运动。

1. 技术概述​

1.1 Hopenet 模型简介​

Hopenet 是一个基于 ResNet 的头部姿态估计网络，能够从单张人脸图像中预测头部的三维旋转角度：

  • Yaw（偏航角）：头部左右转动的角度（-99° ~ +99°）
  • Pitch（俯仰角）：头部上下转动的角度（-99° ~ +99°）
  • Roll（翻滚角）：头部倾斜的角度（-99° ~ +99°）

1.2 应用场景​

  • 🤖 智能跟随：机械臂跟随人脸移动
  • 👁️ 视线追踪：分析用户注视方向
  • 😊 表情分析：辅助表情识别系统
  • 🎮 人机交互：基于头部姿态的控制系统

1.3 系统架构​

3. ONNX模型导出​

3.1 创建导出脚本​

新建 export_onnx.py 文件：

import torch
import torch.nn.functional as F
import torchvision
from hopenet import Hopenet

def export_hopenet_to_onnx():
    """将 Hopenet 模型导出为 ONNX 格式"""
    
    # 1. 构造网络（使用 ResNet50 作为骨干网络，输出 66 个类别）
    model = Hopenet(
        torchvision.models.resnet.Bottleneck,  # 基础模块
        [3, 4, 6, 3],                        # ResNet 层配置
        66                                      # 输出类别数（yaw/pitch/roll 各 22 个 bin）
    )
    
    # 2. 加载预训练权重
    state_dict = torch.load('model/hopenet_robust_alpha1.pkl', map_location='cpu')
    model.load_state_dict(state_dict)
    model.eval()
    print("Model loaded successfully!")
    
    # 3. 准备输入张量（224x224 RGB 人脸图像）
    dummy_input = torch.randn(1, 3, 224, 224)
    
    # 4. 导出 ONNX
    torch.onnx.export(
        model,
        dummy_input,
        "hopenet.onnx",
        export_params=True,              # 导出模型参数
        opset_version=11,              # ONNX 操作集版本
        do_constant_folding=True,       # 常量折叠优化
        input_names=['input'],         # 输入张量名称
        output_names=['yaw', 'pitch', 'roll'],  # 输出张量名称
        dynamic_axes={
            'input': {0: 'batch'},
            'yaw': {0: 'batch'},
            'pitch': {0: 'batch'},
            'roll': {0: 'batch'}
        }
    )
    print("Export hopenet.onnx success!")

if __name__ == '__main__':
    export_hopenet_to_onnx()

4. RKNN模型导出​

4.1 创建转换脚本​

新建 onnx2rknn.py 文件：

import sys
from rknn.api import RKNN

# 配置参数
DATASET_PATH = './dataset.txt'           # 校准图片列表路径
DEFAULT_RKNN_PATH = './hopenet_i8.rknn'  # 输出路径
DEFAULT_QUANT = True                    # 默认 INT8 量化
DEFAULT_PLATFORM = 'rk3576'             # 目标平台

def parse_args():
    """解析命令行参数"""
    if len(sys.argv) < 2:
        print('Usage: python onnx2rknn.py <onnx_path> [platform] [i8|fp] [output_rknn_path]')
        print('        platform: rk3562/rk3566/rk3568/rk3576/rk3588 (default: rk3576)')
        print('        i8|fp: i8=量化, fp=不量化 (default: i8)')
        exit(1)
    
    onnx_path = sys.argv[1]
    platform = sys.argv[2] if len(sys.argv) > 2 else DEFAULT_PLATFORM
    quant_type = sys.argv[3] if len(sys.argv) > 3 else 'i8'
    rknn_path = sys.argv[4] if len(sys.argv) > 4 else DEFAULT_RKNN_PATH
    
    do_quant = quant_type == 'i8'
    return onnx_path, platform, do_quant, rknn_path

if __name__ == '__main__':
    onnx_path, platform, do_quant, rknn_path = parse_args()
    
    # 创建 RKNN 对象
    rknn = RKNN(verbose=False)
    
    # 配置模型参数（使用 ImageNet 预训练模型的均值和标准差）
    print('--> Config model')
    rknn.config(
        mean_values=[[123.675, 116.28, 103.53]],      # BGR 通道均值
        std_values=[[58.395, 57.12, 57.375]],        # BGR 通道标准差
        target_platform=platform                        # 目标平台
    )
    print('done')
    
    # 加载 ONNX 模型
    print('--> Loading ONNX')
    ret = rknn.load_onnx(model=onnx_path)
    if ret != 0:
        print('Load failed!')
        exit(ret)
    print('done')
    
    # 构建 RKNN 模型
    print('--> Building model (do_quant={})'.format(do_quant))
    ret = rknn.build(do_quantization=do_quant, dataset=DATASET_PATH)
    if ret != 0:
        print('Build failed!')
        exit(ret)
    print('done')
    
    # 导出 RKNN 模型
    print('--> Export RKNN')
    ret = rknn.export_rknn(rknn_path)
    if ret != 0:
        print('Export failed!')
        exit(ret)
    print('RKNN saved ->', rknn_path)
    
    # 释放资源
    rknn.release()

# 运行命令
# INT8 量化版本（推荐，速度快）

# FP32 版本（精度高，速度慢）

5. 端侧推理实现​

5.1 完整推理代码​

import cv2
import numpy as np
from rknnlite.api import RKNNLite

# 配置参数
RKNN_PATH = 'hopenet_i8.rknn'     # RKNN 模型路径
IMG_PATH = 'test.jpg'            # 测试图像路径

def softmax(x):
    """Softmax 激活函数"""
    exp = np.exp(x - np.max(x))  # 减去最大值提高数值稳定性
    return exp / exp.sum(axis=-1, keepdims=True)

def draw_axes(img, yaw, pitch, roll, tdx=None, tdy=None, size=80):
    """
    在图像上绘制三维坐标轴（表示头部姿态）
    
    参数:
        img: 输入图像
        yaw: 偏航角（左右转）
        pitch: 俯仰角（上下转）
        roll: 翻滚角（倾斜）
        tdx, tdy: 绘制中心点，默认使用图像中心
        size: 箭头长度
    返回:
        绘制了坐标轴的图像
    """
    h, w = img.shape[:2]
    
    # 使用图像中心作为原点
    if tdx is None or tdy is None:
        tdx, tdy = w // 2, h // 2
    
    # 将角度转换为弧度
    pitch_rad, yaw_rad, roll_rad = map(np.deg2rad, [pitch, yaw, roll])
    
    # 构建旋转矩阵（Rodrigues 公式）
    rotation_vector = np.array([pitch_rad, yaw_rad, roll_rad])
    R, _ = cv2.Rodrigues(rotation_vector)
    
    # 定义 3D 坐标轴端点
    axis = np.float32([
        [size, 0, 0],   # X 轴（红色）
        [0, size, 0],   # Y 轴（绿色）
        [0, 0, size]    # Z 轴（蓝色）
    ])
    
    # 虚拟相机内参矩阵
    K = np.array([
        [1000, 0, tdx],
        [0, 1000, tdy],
        [0, 0, 1]
    ], dtype=np.float32)
    
    # 投影 3D 点到 2D 图像
    axis_2d, _ = cv2.projectPoints(axis, R, np.zeros(3), K, np.zeros(5))
    axis_2d = axis_2d.reshape(-1, 2).astype(int)
    
    # 绘制箭头
    cv2.arrowedLine(img, (tdx, tdy), tuple(axis_2d[0]), (0, 0, 255), 3)   # X 红
    cv2.arrowedLine(img, (tdx, tdy), tuple(axis_2d[1]), (0, 255, 0), 3)   # Y 绿
    cv2.arrowedLine(img, (tdx, tdy), tuple(axis_2d[2]), (255, 0, 0), 3)   # Z 蓝
    
    return img

def decode_angles(yaw, pitch, roll):
    """
    解码角度预测
    
    Hopenet 输出 22 个 bin 的概率分布，通过加权平均计算最终角度
    范围 [-99°, +99°]
    """
    # 创建索引数组：[0, 1, 2, ..., 21]
    idx = np.arange(66)
    
    # 加权求和计算期望值
    yaw_angle = (yaw[0] * idx).sum() * 3 - 99
    pitch_angle = (pitch[0] * idx).sum() * 3 - 99
    roll_angle = (roll[0] * idx).sum() * 3 - 99
    
    return yaw_angle, pitch_angle, roll_angle

def main():
    """主函数：演示 Hopenet 人脸姿态估计"""
    
    # 1. 加载 RKNN 模型
    print('--> Loading RKNN model')
    rknn = RKNNLite()
    ret = rknn.load_rknn(RKNN_PATH)
    if ret != 0:
        print('Load RKNN failed!')
        exit(ret)
    
    # 2. 初始化运行时
    print('--> Init runtime')
    ret = rknn.init_runtime()
    if ret != 0:
        print('Init runtime failed!')
        exit(ret)
    print('done')
    
    # 3. 读取图像预处理
    print('--> Preprocessing image')
    img = cv2.imread(IMG_PATH)                # BGR
    img = cv2.resize(img, (224, 224))          # 缩放到 224x224
    img = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)  # BGR 转 RGB
    img = np.expand_dims(img, 0)                # NHWC
    
    # 4. 推理
    print('--> Running inference')
    outputs = rknn.inference(inputs=[img])
    
    # 5. Softmax 归一化
    yaw, pitch, roll = [softmax(out[0]) for out in outputs]
    
    # 6. 解码角度
    yaw_angle, pitch_angle, roll_angle = decode_angles(yaw, pitch, roll)
    print(f'yaw={yaw_angle:6.2f}°  pitch={pitch_angle:6.2f}°  roll={roll_angle:6.2f}°')
    
    # 7. 可视化
    print('--> Visualizing')
    vis = cv2.imread(IMG_PATH)
    vis = draw_axes(vis, yaw_angle, pitch_angle, roll_angle)
    cv2.imwrite('result.jpg', vis)
    print('Saved to result.jpg')
    
    # 8. 释放资源
    rknn.release()
    print('Done!')

if __name__ == '__main__':
    main()

6. 跟随控制实现​

6.1 角度控制逻辑​

class FaceFollowingController:
    """人脸跟随控制器"""
    
    def __init__(self, robot, deadzone_yaw=10, deadzone_pitch=10):
        self.robot = robot
        self.deadzone_yaw = deadzone_yaw      # 偏航角死区
        self.deadzone_pitch = deadzone_pitch    # 俯仰角死区
        self.kp = 0.5                           # 比例增益
        
        # 平滑滤波器
        self.ema_alpha = 0.3
        self.last_yaw = 0
        self.last_pitch = 0
    
    def ema_filter(self, current, previous):
        """指数移动平均滤波"""
        return self.ema_alpha * current + (1 - self.ema_alpha) * previous
    
    def calculate_control(self, yaw, pitch):
        """计算控制信号"""
        # 平滑滤波
        yaw_smoothed = self.ema_filter(yaw, self.last_yaw)
        pitch_smoothed = self.ema_filter(pitch, self.last_pitch)
        self.last_yaw = yaw_smoothed
        self.last_pitch = pitch_smoothed
        
        # 应用死区（防止抖动）
        if abs(yaw_smoothed) < self.deadzone_yaw:
            yaw_smoothed = 0
        if abs(pitch_smoothed) < self.deadzone_pitch:
            pitch_smoothed = 0
        
        # 比例控制
        pan_control = self.kp * yaw_smoothed
        tilt_control = self.kp * pitch_smoothed
        
        return pan_control, tilt_control
    
    def control(self, yaw, pitch):
        """执行控制"""
        pan, tilt = self.calculate_control(yaw, pitch)
        
        # 发送到机械臂
        self.robot.move_pan(pan)
        self.robot.move_tilt(tilt)
        
        return pan, tilt

class FaceFollowingSystem:
    """人脸跟随系统"""
    
    def __init__(self, det_model, pose_model, robot):
        # 加载检测模型
        self.det_rknn = RKNNLite()
        self.det_rknn.load_rknn(det_model)
        self.det_rknn.init_runtime()
        
        # 加载姿态模型
        self.pose_rknn = RKNNLite()
        self.pose_rknn.load_rknn(pose_model)
        self.pose_rknn.init_runtime()
        
        # 控制器
        self.controller = FaceFollowingController(robot)
        self.tracking_enabled = False
    
    def process_frame(self, frame):
        """处理单帧图像"""
        if not self.tracking_enabled:
            return frame
        
        # 1. 人脸检测
        faces = self.detect_faces(frame)
        
        if len(faces) == 0:
            return frame
        
        # 2. 选择最大的人脸
        face = max(faces, key=lambda f: f[2] * f[3])
        x1, y1, x2, y2 = face[:4]
        
        # 3. 提取人脸区域
        face_roi = frame[y1:y2, x1:x2]
        if face_roi.size == 0:
            return frame
        
        # 4. 姿态估计
        yaw, pitch, roll = self.estimate_pose(face_roi)
        
        # 5. 控制机械臂
        self.controller.control(yaw, pitch)
        
        # 6. 可视化
        vis = frame.copy()
        cv2.rectangle(vis, (x1, y1), (x2, y2), (0, 255, 0), 2)
        
        cx, cy = (x1 + x2) // 2, (y1 + y2) // 2
        draw_axes(vis, yaw, pitch, roll, cx, cy)

        face_resized = cv2.resize(face_roi, (224, 224))
        face_rgb = cv2.cvtColor(face_resized, cv2.COLOR_BGR2RGB)
        face_input = np.expand_dims(face_rgb, 0)
        
        # 推理

7. 常见问题与解决​

7.1 人脸检测失败​

可能原因：光照条件差、人脸过小或被遮挡

解决方案：降低检测阈值、增强图像亮度、扩大搜索区域

7.2 姿态估计不准确​

可能原因：人脸框不准确、裁剪过多

解决方案：扩大人脸框、增加 padding、应用数据增强

7.3 机械臂抖动​

解决方案：增大死区、降低增益、增大平滑系数
